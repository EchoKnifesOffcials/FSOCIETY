--[[ MM2 TRADE BOT - WORKING WITH REAL REMOTES Uses actual MM2 trade remote names Works on Synapse X / ScriptWare ]] local player = game:GetService("Players").LocalPlayer local replicatedStorage = game:GetService("ReplicatedStorage") local chatService = game:GetService("Chat") local userInputService = game:GetService("UserInputService") -- ===== TRADE REMOTES ===== local Trade = replicatedStorage:FindFirstChild("Trade") if not Trade then warn("Trade folder not found!") return end local remotes = { getRequest = Trade:FindFirstChild("GetRequest"), acceptRequest = Trade:FindFirstChild("AcceptRequest"), declineRequest = Trade:FindFirstChild("DeclineRequest"), acceptTrade = Trade:FindFirstChild("AcceptTrade"), declineTrade = Trade:FindFirstChild("DeclineTrade"), startTrade = Trade:FindFirstChild("StartTrade"), offerItem = Trade:FindFirstChild("OfferItem"), updateOffers = Trade:FindFirstChild("UpdateOffers"), removeOffer = Trade:FindFirstChild("RemoveOffer"), endTrade = Trade:FindFirstChild("EndTrade"), cancelRequest = Trade:FindFirstChild("CancelRequest"), completeTrade = Trade:FindFirstChild("CompleteTrade"), updateTrade = Trade:FindFirstChild("UpdateTrade"), cancelAccept = Trade:FindFirstChild("CancelAccept"), setRequestsEnabled = Trade:FindFirstChild("SetRequestsEnabled"), requestSent = Trade:FindFirstChild("RequestSent") } -- Check all remotes exist for name, remote in pairs(remotes) do if not remote then print("[WARN] Missing remote: " .. name) else print("[OK] Found: " .. name) end end -- ===== BOT STATE ===== local botActive = true local tradeCount = 0 local currentTradePartner = nil local currentOffers = {} -- ===== FORCE TABLET UI ===== userInputService.TouchEnabled = true userInputService.MouseEnabled = false -- ===== SEND TRADE REQUEST ===== local function sendTradeRequest(username) -- Find target player local targetPlayer = nil for _, plr in pairs(game:GetService("Players"):GetPlayers()) do if plr.Name:lower() == username:lower() then targetPlayer = plr break end end if not targetPlayer then print("[❌] Player not found: " .. username) return false end -- Method 1: Use GetRequest remote if remotes.getRequest and remotes.getRequest:IsA("RemoteEvent") then remotes.getRequest:FireServer(targetPlayer) print("[📤] Trade request sent to: " .. username) currentTradePartner = targetPlayer return true end -- Method 2: Use StartTrade remote if remotes.startTrade and remotes.startTrade:IsA("RemoteEvent") then remotes.startTrade:FireServer(targetPlayer) print("[📤] Trade started with: " .. username) currentTradePartner = targetPlayer return true end print("[❌] No working trade remote found") return false end -- ===== ACCEPT TRADE REQUEST ===== local function acceptTradeRequest() if remotes.acceptRequest and remotes.acceptRequest:IsA("RemoteEvent") then remotes.acceptRequest:FireServer() print("[✅] Trade request accepted") return true end return false end -- ===== DECLINE TRADE REQUEST ===== local function declineTradeRequest() if remotes.declineRequest and remotes.declineRequest:IsA("RemoteEvent") then remotes.declineRequest:FireServer() print("[❌] Trade request declined") return true end return false end -- ===== OFFER ITEM ===== local function offerItem(itemId, itemData) if remotes.offerItem and remotes.offerItem:IsA("RemoteEvent") then remotes.offerItem:FireServer(itemId, itemData) print("[🎁] Offered item: " .. tostring(itemId)) table.insert(currentOffers, itemId) return true end return false end -- ===== REMOVE OFFER ===== local function removeOffer(itemId) if remotes.removeOffer and remotes.removeOffer:IsA("RemoteEvent") then remotes.removeOffer:FireServer(itemId) print("[🗑️] Removed offer: " .. tostring(itemId)) return true end return false end -- ===== ACCEPT TRADE (Final accept) ===== local function acceptTrade() if remotes.acceptTrade and remotes.acceptTrade:IsA("RemoteEvent") then remotes.acceptTrade:FireServer() print("[✅] Trade accepted!") return true end return false end -- ===== DECLINE TRADE ===== local function declineTrade() if remotes.declineTrade and remotes.declineTrade:IsA("RemoteEvent") then remotes.declineTrade:FireServer() print("[❌] Trade declined") return true end return false end -- ===== CANCEL REQUEST ===== local function cancelRequest() if remotes.cancelRequest and remotes.cancelRequest:IsA("RemoteEvent") then remotes.cancelRequest:FireServer() print("[🔄] Trade request cancelled") return true end return false end -- ===== END TRADE ===== local function endTrade() if remotes.endTrade and remotes.endTrade:IsA("RemoteEvent") then remotes.endTrade:FireServer() print("[🏁] Trade ended") return true end return false end -- ===== COMPLETE TRADE ===== local function completeTrade() if remotes.completeTrade and remotes.completeTrade:IsA("RemoteEvent") then remotes.completeTrade:FireServer() tradeCount = tradeCount + 1 print("[🎉] Trade completed! Total: " .. tradeCount) return true end return false end -- ===== UPDATE OFFERS ===== local function updateOffers(offersTable) if remotes.updateOffers and remotes.updateOffers:IsA("RemoteEvent") then remotes.updateOffers:FireServer(offersTable) print("[🔄] Offers updated") return true end return false end -- ===== SET REQUESTS ENABLED ===== local function setRequestsEnabled(enabled) if remotes.setRequestsEnabled and remotes.setRequestsEnabled:IsA("RemoteEvent") then remotes.setRequestsEnabled:FireServer(enabled) print("[⚙️] Requests enabled: " .. tostring(enabled)) return true end return false end -- ===== COMPLETE TRADE SEQUENCE ===== local function doFullTrade(username, items) if not botActive then print("[⚠️] Bot is paused") return false end print("[🤖] Starting trade with: " .. username) -- Step 1: Send trade request local sent = sendTradeRequest(username) if not sent then print("[❌] Failed to send request") return false end -- Step 2: Wait for player to accept request print("[⏳] Waiting for player to accept request...") task.wait(5) -- Step 3: Accept trade (if needed) acceptTrade() -- Step 4: Offer items task.wait(1) for _, item in pairs(items) do offerItem(item, {name = item}) task.wait(0.3) end -- Step 5: Wait for player to ready up print("[⏳] Waiting for player to ready...") task.wait(8) -- Step 6: Accept trade acceptTrade() task.wait(1) -- Step 7: Complete trade completeTrade() -- Step 8: Send confirmation message task.wait(2) pcall(function() chatService:Chat(player.Character.Head, "Trade complete! ✅", Enum.ChatColor.Green) end) print("[✅] Trade completed with: " .. username) return true end -- ===== AUTO ACCEPT INCOMING REQUESTS ===== local function autoAcceptIncoming() -- Hook into RequestSent remote if remotes.requestSent and remotes.requestSent:IsA("RemoteEvent") then remotes.requestSent.OnClientEvent:Connect(function(sender) print("[📨] Trade request from: " .. sender.Name) task.wait(1) acceptTradeRequest() print("[✅] Auto-accepted request from: " .. sender.Name) end) end -- Hook into UpdateTrade for real-time updates if remotes.updateTrade and remotes.updateTrade:IsA("RemoteEvent") then remotes.updateTrade.OnClientEvent:Connect(function(data) print("[🔄] Trade update received") end) end end -- ===== CHAT COMMAND MONITOR ===== local function monitorChat() pcall(function() local chatFrame = player.PlayerGui:FindFirstChild("Chat") if chatFrame then local chatLog = chatFrame:FindFirstChild("ChatLog") if chatLog then chatLog.ChildAdded:Connect(function(msg) task.wait(0.5) if msg:IsA("TextLabel") then local text = msg.Text if text and text:lower():find("!trade") then local username = text:match("^(.-):") if username then username = username:gsub("%s+", "") print("[💬] Trade command from: " .. username) task.spawn(function() doFullTrade(username, {"Chroma Knife", "Elderwood Scythe"}) end) end end end end) end end end) end -- ===== CREATE GUI ===== local function status.Font = Enum.Font.Gotham status.Parent = frame -- Username input local nameBox = Instance.new("TextBox") nameBox.Size = UDim2.new(1, -20, 0, 35) nameBox.Position = UDim2.new(0, 10, 0, 100)
