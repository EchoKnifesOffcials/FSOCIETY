checkBtn.Text = "Check Trade" checkBtn.BackgroundColor3 = Color3.fromRGB(0,120,200) -- ===== LOGIC ===== local function calculateTotal(text) local total = 0 for item in text:gmatch("[^,]+") do item = item:gsub("^%s+",""):gsub("%s+$","") if itemValues[item] then total += itemValues[item] end end return total end checkBtn.MouseButton1Click:Connect(function() local yourTotal = calculateTotal(yourBox.Text) local theirTotal = calculateTotal(theirBox.Text) local diff = theirTotal - yourTotal if diff > 0 then result.Text = "✅ WIN (+ "..diff..")" result.TextColor3 = Color3.fromRGB(0,255,0) elseif diff < 0 then result.Text = "❌ LOSS ("..diff..")" result.TextColor3 = Color3.fromRGB(255,0,0) else result.Text = "⚖️ FAIR TRADE" result.TextColor3 = Color3.fromRGB(255,255,0) end end)
