game:GetService("CoreGui") end) end -- Create a simple red frame (VERY VISIBLE) local frame = Instance.new("Frame") frame.Size = UDim2.new(0, 300, 0, 200) frame.Position = UDim2.new(0.5, -150, 0.5, -100) frame.BackgroundColor3 = Color3.fromRGB(255, 0, 0) -- RED so you see it frame.BackgroundTransparency = 0 frame.BorderSizePixel = 5 frame.BorderColor3 = Color3.fromRGB(255, 255, 255) frame.Parent = gui -- Simple title local title = Instance.new("TextLabel") title.Size = UDim2.new(1, 0, 0, 40) title.BackgroundColor3 = Color3.fromRGB(0, 0, 0) title.Text = "TRADE BOT" title.TextColor3 = Color3.fromRGB(255, 255, 255) title.TextSize = 20 title.Font = Enum.Font.GothamBold title.Parent = frame -- Username input local input = Instance.new("TextBox") input.Size = UDim2.new(1, -20, 0, 40) frame -- Send button local button = Instance.new("TextButton") button.Size = UDim2.new(1, -20, 0, 50) button.Position = UDim2.new(0, 10, 0, 100) button.BackgroundColor3 = Color3.fromRGB(0, 100, 200) button.Text = "SEND TRADE" button.TextColor3 = Color3.fromRGB(255, 255, 255) button.TextSize = 16 button.Font = Enum.Font.GothamBold button.Parent = frame -- Status label local status = Instance.new("TextLabel") status.Size = UDim2.new(1, -20, 0, 30) status.Position = UDim2.new(0, 10, 0, 160) status.BackgroundTransparency = 1 status.Text = "Ready" status.TextColor3 = Color3.fromRGB(0, 255, 0) status.TextSize = 12 status.Parent = frame -- Button function button.MouseButton1Click:Connect(function() local username = input.Text if username == "" then status.Text = "..." status.TextColor3 = Color3.fromRGB(255, 255, 0) -- Try to send trade local success = false pcall(function() -- Look for trade button in player list local playerList = player.PlayerGui:FindFirstChild("PlayerList") if playerList then for _, btn in pairs(playerList:GetDescendants()) do if btn:IsA("TextButton") and btn.Name:lower():find("trade") then local parent = btn.Parent if parent then local nameLabel = parent:FindFirstChild("NameLabel") or parent:FindFirstChild("Username") if nameLabel and nameLabel.Text:lower() == username:lower() then btn:Click() success = true status.Text = "Trade sent to ".. username status.TextColor3 = Color3.fromRGB(0, 255, 0) end end end end end end) if not success then status.Text = "Failed - Player not found" status.TextColor3 = Color3.fromRGB(255, 0, 0) end -- Clear after 3 seconds task.wait(3) if status.Text ~= "Ready" then status.Text = "Ready" status.TextColor3 = Color3.fromRGB(0, 255, 0) end end) -- Print to console print("========================================") print("✅ SIMPLE GUI CREATED") print("📍 Look for a RED box in the middle of screen") print("📍 If you see it, enter username and click SEND") print("========================================") -- Flash the frame for i = 1, 10 do frame.BackgroundColor3 = Color3.fromRGB(255, 255, 255) task.wait(0.1) frame.BackgroundColor3 = Color3.fromRGB(255, 0, 0) task.wait(0.1) end
